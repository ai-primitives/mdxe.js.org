# Next.js Plugin API

## withMDXE Function

```typescript
interface MDXEPluginOptions {
  mdxOptions?: Record<string, unknown>
  watch?: {
    enabled?: boolean
    ignore?: string[]
  }
}

function withMDXE(
  nextConfig: NextConfig = {},
  pluginOptions: MDXEPluginOptions = {}
): NextConfig
```

## Configuration Examples

```javascript
// Basic usage
export default withMDXE()

// With options
export default withMDXE({
  // Next.js config
  reactStrictMode: true
}, {
  // MDXE options
  mdxOptions: {
    jsx: true
  },
  watch: {
    enabled: true
  }
})
```
